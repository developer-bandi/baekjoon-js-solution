# 1. 문제

https://www.acmicpc.net/problem/4963

# 2. 아이디어

연결요소의 갯수를 찾는 문제이다. 이 문제의 경우 여러개의 테스트케이스가 존재하며, 테이블을 그래프 형태로 따로 가공할 필요가 없으며, 상하좌우뿐만 아니라 대각선요소까지 고려해야하기 때문에, 8개의 분기를 나누어 bfs를 구성하였다.

# 3. 소스코드

```javascript
const fs = require("fs");
const mapArr = fs.readFileSync("/dev/stdin").toString().trim().split("\n");

let currentIndex = 0;
const resultArr = [];

function findStart(arr) {
  for (let i = 0; i < arr.length; i++) {
    const index = arr[i].indexOf(1);
    if (index !== -1) {
      arr[i][index] = 0;
      return [i, index];
    }
  }
  return false;
}

function BFS(start, table) {
  const queue = [start];
  while (queue.length > 0) {
    const visitNum = queue.shift();
    if (visitNum[0] !== 0 && table[visitNum[0] - 1][visitNum[1]] === 1) {
      queue.push([visitNum[0] - 1, visitNum[1]]);
      table[visitNum[0] - 1][visitNum[1]] = 0;
    }

    if (visitNum[1] !== 0 && table[visitNum[0]][visitNum[1] - 1] === 1) {
      queue.push([visitNum[0], visitNum[1] - 1]);
      table[visitNum[0]][visitNum[1] - 1] = 0;
    }

    if (
      visitNum[0] !== table.length - 1 &&
      table[visitNum[0] + 1][visitNum[1]] === 1
    ) {
      queue.push([visitNum[0] + 1, visitNum[1]]);
      table[visitNum[0] + 1][visitNum[1]] = 0;
    }

    if (
      visitNum[1] !== table[0].length - 1 &&
      table[visitNum[0]][visitNum[1] + 1] === 1
    ) {
      queue.push([visitNum[0], visitNum[1] + 1]);
      table[visitNum[0]][visitNum[1] + 1] = 0;
    }

    if (
      visitNum[0] !== 0 &&
      visitNum[1] !== 0 &&
      table[visitNum[0] - 1][visitNum[1] - 1] === 1
    ) {
      queue.push([visitNum[0] - 1, visitNum[1] - 1]);
      table[visitNum[0] - 1][visitNum[1] - 1] = 0;
    }

    if (
      visitNum[0] !== 0 &&
      visitNum[1] !== table[0].length - 1 &&
      table[visitNum[0] - 1][visitNum[1] + 1] === 1
    ) {
      queue.push([visitNum[0] - 1, visitNum[1] + 1]);
      table[visitNum[0] - 1][visitNum[1] + 1] = 0;
    }
    if (
      visitNum[0] !== table.length - 1 &&
      visitNum[1] !== 0 &&
      table[visitNum[0] + 1][visitNum[1] - 1] === 1
    ) {
      queue.push([visitNum[0] + 1, visitNum[1] - 1]);
      table[visitNum[0] + 1][visitNum[1] - 1] = 0;
    }
    if (
      visitNum[0] !== table.length - 1 &&
      visitNum[1] !== table[0].length - 1 &&
      table[visitNum[0] + 1][visitNum[1] + 1] === 1
    ) {
      queue.push([visitNum[0] + 1, visitNum[1] + 1]);
      table[visitNum[0] + 1][visitNum[1] + 1] = 0;
    }
  }
}

function solution(table) {
  let start = findStart(table);
  let result = 0;
  while (start) {
    BFS(start, table);
    result++;
    start = findStart(table);
  }
  return result;
}

while (true) {
  const [w, h] = mapArr[currentIndex++].split(" ").map(Number);
  if (w === 0 && h === 0) break;
  const table = [];
  for (let j = currentIndex; j < currentIndex + h; j++) {
    table.push(mapArr[j].split(" ").map(Number));
  }
  resultArr.push(solution(table));
  currentIndex += h;
}

console.log(resultArr.join("\n"));
```
