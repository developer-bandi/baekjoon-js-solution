# 1. 문제

https://www.acmicpc.net/problem/1012

# 2. 아이디어

2667 단지번호 붙이기와 유사하면서, 그래프 정점의 갯수를 세지 않고, 총개수만을 출력하면 되는 문제이기에 약간 쉽지만 입력값을 표 형태로 가공해야 하기 때문에 결국 난이도는 유사하다.

주어진 테스트케이스에 대해서 입력값을 테이블 형태로 가공한뒤, 시작지점을 임의로 찾아 bfs로 탐색하는 과정을 반복하여 테스트케이스 별로 연결된 그래프의 갯수가 몇개인지 찾아내면 된다.

# 3. 소스코드

```javascript
const fs = require("fs");
const [T, ...landArr] = fs
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

let currentIndex = 0;
const resultArr = [];

function findStart(arr) {
  for (let i = 0; i < arr.length; i++) {
    const index = arr[i].indexOf(1);
    if (index !== -1) {
      arr[i][index] = 0;
      return [i, index];
    }
  }
  return false;
}

function BFS(start, table) {
  const queue = [start];
  while (queue.length > 0) {
    const visitNum = queue.shift();
    if (visitNum[0] !== 0 && table[visitNum[0] - 1][visitNum[1]] === 1) {
      queue.push([visitNum[0] - 1, visitNum[1]]);
      table[visitNum[0] - 1][visitNum[1]] = 0;
    }

    if (visitNum[1] !== 0 && table[visitNum[0]][visitNum[1] - 1] === 1) {
      queue.push([visitNum[0], visitNum[1] - 1]);
      table[visitNum[0]][visitNum[1] - 1] = 0;
    }

    if (
      visitNum[0] !== table.length - 1 &&
      table[visitNum[0] + 1][visitNum[1]] === 1
    ) {
      queue.push([visitNum[0] + 1, visitNum[1]]);
      table[visitNum[0] + 1][visitNum[1]] = 0;
    }

    if (
      visitNum[1] !== table[0].length - 1 &&
      table[visitNum[0]][visitNum[1] + 1] === 1
    ) {
      queue.push([visitNum[0], visitNum[1] + 1]);
      table[visitNum[0]][visitNum[1] + 1] = 0;
    }
  }
}

function solution(table) {
  let start = findStart(table);
  let result = 0;
  while (start) {
    BFS(start, table);
    result++;
    start = findStart(table);
  }
  return result;
}

for (let i = 0; i < Number(T); i++) {
  const [M, N, K] = landArr[currentIndex++].split(" ").map(Number);
  const table = [];
  for (let j = 0; j < N; j++) {
    table[j] = new Array(M).fill(0);
  }
  for (let j = currentIndex; j < currentIndex + K; j++) {
    const position = landArr[j].split(" ").map(Number);
    table[position[1]][position[0]] = 1;
  }
  resultArr.push(solution(table));
  currentIndex += K;
}

console.log(resultArr.join("\n"));
```
